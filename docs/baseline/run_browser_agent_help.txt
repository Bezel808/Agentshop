options:
  -h, --help            show this help message and exit
  --api-key API_KEY     API Key
  --llm {openai,qwen,kimi}
                        LLM backend
  --temperature TEMPERATURE
                        LLM 采样温度
  --perception {visual,verbal}
                        感知模式: visual=截图, verbal=文本
  --query QUERY         用户购物需求
  --server SERVER       Web 服务器 URL
  --condition-name CONDITION_NAME
                        实验条件名
  --page PAGE           初始页码
  --price-min PRICE_MIN
                        价格下限
  --price-max PRICE_MAX
                        价格上限
  --rating-min RATING_MIN
                        最低星级
  --once                完成一次后退出
  --legacy-fallback     启用旧版文本猜工具 fallback
  --trace-output TRACE_OUTPUT
                        JSONL trace 输出路径
  --env-backend {auto,api,playwright,mcp}
                        环境后端，默认 auto: visual->playwright, verbal->api
  --mcp-endpoint MCP_ENDPOINT
                        MCP HTTP endpoint, required for --env-backend mcp
